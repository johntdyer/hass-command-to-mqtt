#include <tunables/global>

profile ssh-command-mqtt flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within data path
  /data/** rw,

  # Start new profile
  /usr/bin/ssh-command-mqtt cx -> ssh-command-mqtt,

  profile ssh-command-mqtt flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) set=(kill,term,int,hup,cont),

    # Access to options.json and other files within data path
    /data/** rw,

    # Access to SSL certificates
    /ssl/** r,

    # Python
    /usr/bin/python3 ix,
    /usr/lib/python3*/** r,

    # SSH client
    /usr/bin/ssh rix,
    /etc/ssh/** r,
    /root/.ssh/** rw,

    # Network access
    network inet dgram,
    network inet stream,

    # Process control
    signal (send) set=(kill,term,int,hup,cont),

    # Temporary files
    /tmp/** rw,
    /var/tmp/** rw,
  }
}
